);

CREATE TRIGGER trg_after_insert_pilotos
ON tb_pilotos
AFTER INSERT
AS
BEGIN
    UPDATE tb_estadisticas
    SET total_victorias = (
        SELECT SUM(num_victorias)
        FROM tb_pilotos
    );
END;
